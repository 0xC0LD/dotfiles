# make swap 8gb (ram size), set Type to Linux Swap
# the rest of the space is the system, set Type to Linux, set bootable

mkswap /dev/sdx1
swapon /dev/sdx1
mkfs.ext4 /dev/sdx2

===[ INSTALL ARCH LINUX ]===
mount /dev/sdx2 /mnt
pacstrap /mnt base linux linux-firmware
